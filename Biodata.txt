    private void jCheckBox1StateChanged(javax.swing.event.ChangeEvent evt) {                                        
        // TODO add your handling code here:
        if (jCheckBox1.isSelected())
        { jCheckBox1.setText("Menikah");

        }
        else {jCheckBox1.setText("Belum Menikah");
            status = "Belum Menikah";
        }
    }                                       

    private void jButton1MouseClicked(java.awt.event.MouseEvent evt) {                                      
        // TODO add your handling code here:
        String namaMhs = jTextField1.getText();
        String nim = jTextField2.getText();
        String tm = jTextField3.getText();
        String alamat = jTextField4.getText();
        String asal = jTextField5.getText();
        if (jRadioButton1.isSelected()){
            jk = jRadioButton1.getText();
        }
        else if (jRadioButton2.isSelected()) {
            jk = jRadioButton2.getText();
        }
        if (jComboBox1.getSelectedItem()== "Islam") {
            agama = "Islam";
        }
        else if (jComboBox1.getSelectedItem()== "Kristen") {
            agama = "Kristen";
        }
        else if (jComboBox1.getSelectedItem()== "Katolik") {
            agama = "Katolik";
        }
        else if (jComboBox1.getSelectedItem()== "Buda"){
            agama = "Buda";
            
        }
        else if (jComboBox1.getSelectedItem()== "Hindu") {
            agama = "Hindu";
        }
        jTextArea1.setText("\nNama Mahasiswa : "+namaMhs+"\nNim\t     : "+nim+"\nTahun Masuk\t     : "+
                tm+"\nJenis Kelamin\t     : "+jk+"\nAgama\t     : "+agama+"\nAlamat\t     : "+alamat+"\n"
                        + "Daerah Asal\t     : "+asal+"\nStatus\t     : "+status);
    }
